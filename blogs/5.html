    /**
 * common enum hooks

 import { storeToRefs } from 'pinia'
 import { useCommonStore } from '@/store/common'

 export const useEnum = () => {
   const commonStore = useCommonStore()
   const {
     statusEnum,
     currencyEnum,
     commonEnum
   } = storeToRefs(commonStore)
   const {
     getStatusEnum,
     getCurrencyEnum,
     getCommonEnum
   } = commonStore

   /**
    * page onMounted check enum exists
    * if not exists call effect functions
    * @param {array} arr need enum array
    * @return {void}
    */
   const initEnum = (arr: Array<string>, enumList: Array<string>) => {
     if (!arr.length) return

     for (let i = 0; i < arr.length; i++) {
       switch (arr[i]) {
         case 'statusEnum': {
           if (!statusEnum.value.length) {
             getStatusEnum()
           }
           break
         }
         case 'currencyEnum': {
          if (!currencyEnum.value.length) {
            getCurrencyEnum()
          }
          break
        }
         case 'commonEnum': {
          const needList = enumList.filter((item: string) => !commonEnum.value[item] || !commonEnum.value[item].length)
          if (needList.length) {
            getCommonEnum(needList)
          }
          break
         }
         default:
           break
       }
     }
   }

   return {
     statusEnum,
     currencyEnum,
     commonEnum,
     initEnum
   }
 }

    import { reactive, toRefs } from 'vue'
import { defineStore } from 'pinia'
import * as Api from '@/api/common'
import type { IStatusItem, ICurrencyItem, ICommonEnumItem } from '@/model/commonModel'

interface ICommon {
  statusEnum: Array<IStatusItem>
  currencyEnum: Array<ICurrencyItem>
  commonEnum: {
    [key: string]: Array<ICommonEnumItem>
  }
}

const useCommonEnumEffect = () => {
  const common: ICommon = reactive({
    statusEnum: [],
    currencyEnum: [],
    commonEnum: {}
  })

  const getStatusEnum = async () => {
    try {
      const { data } = await Api.getStatusList<Array<IStatusItem>>()

      if (data && data.length) {
        const filteredData = data.filter(item => item.itemCode !== 'A')
        // filteredData.unshift({ itemCode: '', itemName: 'å…¨éƒ¨' })
        common.statusEnum = filteredData
      }
    } catch (err) {
      console.error(err)
    }
  }

  const getCurrencyEnum = async () => {
    try {
      const { data } = await Api.getCurrencyList<Array<ICurrencyItem>>()

      if (data && data.length) {
        common.currencyEnum = data
      }
    } catch (err) {
      console.error(err)
    }
  }

  const getCommonEnum = async (enumList: Array<string>) => {
    try {
      const { data } = await Api.getCommonEnum(enumList)

      if (data && Object.keys(data).length > 0) {
        common.commonEnum = {
          ...common.commonEnum,
          ...data
        }
      }
    } catch (err) {
      console.error(err)
    }
  }

  return {
    common,
    getStatusEnum,
    getCurrencyEnum,
    getCommonEnum
  }
}

export const useCommonStore = defineStore('common', () => {
  const {
    common,
    getStatusEnum,
    getCurrencyEnum,
    getCommonEnum,
  } = useCommonEnumEffect()

  return {
    ...toRefs(common),
    getStatusEnum,
    getCurrencyEnum,
    getCommonEnum
  }
})

    import { ref } from 'vue'
import Stomp from 'stompjs'

import request from '@/utils/request'

  TOKEN,
  TENANT_ID,
  SOCKET

import type { Ref } from 'vue'
import type { Client } from 'stompjs'

export const useWebSocketExport = (socketTime: Ref, emitter: any = null) => {
  const stompClient = ref<Client | null>(null)
  const downloaded = ref(false)
  const websocketTimer = ref<NodeJS.Timer>()
  const { downloadFn } = useExportFile()

  /**
   * init websocket
   * @return {void}
   */
  const initWebSocket = () => {
    const tenantId = cookie.get(TENANT_ID) || 1
    if (typeof WebSocket === 'undefined') {
      message('æ‚¨çš„æµè§ˆå™¨ä¸æ”¯æŒ WebSocket')
    } else {
      if (stompClient.value === null || stompClient.value.connected) {
        const socket = new SockJS(import.meta.env.VITE_APP_EXPORT_SOCKET_ENDPOINT)

        stompClient.value = Stomp.over(socket)
        stompClient.value.heartbeat.outgoing = 20000
        stompClient.value.heartbeat.incoming = 0
        const headers: { [key: string]: any } = {}
        const token = cookie.get(TOKEN)
        headers.fauthorization = token

        stompClient.value.connect(headers, () => {
          subscribeUser(+tenantId)
        }, () => {
          console.log('connect failed')
        })
      }
    }
  }

  const subscribeUser = (tenantId: number) => {
    const destination = `${import.meta.env.VITE_APP_EXPORT_SUBSCRIBE_PREFIX}/${tenantId}${import.meta.env.VITE_APP_EXPORT_SUBSCRIBE}`

    console.log(`/user${destination}`)

    stompClient.value!.subscribe(`/user${destination}`, (res: any) => {
      const { code, data } = JSON.parse(res.body)

      console.log(code, data)
      if (code === '0' && data.type === 'download' && !downloaded.value) {
        const getSockTime = storage.get(SOCKET)
        if (getSockTime === socketTime.value) {
          exportFile(data.content)
        }
      }
    })
  }

  /**
   * export
   * @param {string} url
   * @return {void}
   */
  const exportFile = async (url: string) => {
    if (websocketTimer.value) {
      clearInterval(websocketTimer.value)
    }

    const filename = url.substring(url.lastIndexOf('=') + 1)
    const requestId = url.substring(url.lastIndexOf('/') + 1, url.lastIndexOf('?'))
    const token = cookie.get(TOKEN)

    const { data } = await request({
      url,
      method: 'get',
      responseType: 'blob',
      headers: { fauthorization: token }
    })

    const href = window.URL.createObjectURL(data)
    await downloadFn({ href, filename })
    await updateExportStatus(requestId)
  }

  /**
   * update status
   * @param {string} requestId
   */
  const updateExportStatus = async (requestId: string) => {
    const params = {
      requestId,
      status: 'C'
    }

    const { data } = await Api.updateExportStatus(params)
    // return data
    if (data === 0) {
      message('ä¸‹è½½å¯¼å‡ºæ–‡ä»¶å®Œæˆ', 'success')
      downloaded.value = true
      emitter && emitter.emit('endExportLoading')
    }
  }

  /**
  * close websocket
  */
  const closeWebSocket = () => {
    if (stompClient.value !== null) {
      // eslint-disable-next-line @typescript-eslint/no-empty-function
      stompClient.value!.disconnect(() => { })
    }
  }

  /**
  * Ask the backend actively every half minute
  * @param {string} requestId
  */
  const askExportStatus =  (requestId: string) => {
    websocketTimer.value = setInterval(async () => {
      const { data: { status, url } } = await Api.askExportStatus(requestId)
      if (['C', 'D'].includes(status.toUpperCase()) && !downloaded.value) {
        exportFile(url)
      } else if (['E'].includes(status.toUpperCase())) {
        message('å¯¼å‡ºå¼‚å¸¸ï¼Œè¯·è”ç³»é‚®ç®±ï¼šoa_service@qiyi.com çƒ­èŠï¼šåæ§½å›')
        clearInterval(websocketTimer.value)
        emitter && emitter.emit('endExportLoading')
      }
    }, 10000)
  }

  const rollbackDownloadStatus = () => {
    downloaded.value = false
  }

  return {
    initWebSocket,
    closeWebSocket,
    askExportStatus,
    rollbackDownloadStatus
  }
}
