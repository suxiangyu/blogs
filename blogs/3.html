    handlePicture(editor) {
      const observer = new MutationObserver((mutationsList) => {
        for (let mutation of mutationsList) {
          if (mutation.type === 'childList') {
            mutation.addedNodes.forEach(node => {
              if (node.nodeName.toLowerCase() === 'img') {
                this.ensureImageLoadedAndResize(node);
              } else if (node.querySelectorAll) {
                Array.from(node.querySelectorAll('img')).forEach(img => {
                  this.ensureImageLoadedAndResize(img);
                });
              }
            });
          }
        }
      });

      editor.on('init', () => {
        const editorBody = editor.getBody();
        observer.observe(editorBody, { childList: true, subtree: true });
      });
    },
    ensureImageLoadedAndResize(imgElement) {
      const maxWidth = 100;

      const handleImageLoad = () => {
        let newWidth = imgElement.naturalWidth;
        let newHeight = imgElement.naturalHeight;

        if (imgElement.naturalWidth > maxWidth) {
          newWidth = maxWidth;
          newHeight = (imgElement.naturalHeight / imgElement.naturalWidth) * maxWidth;
        }

        // æ›´æ–°å›¾åƒå®½é«˜å’Œæ ·å¼

        imgElement.setAttribute('height', newHeight);
