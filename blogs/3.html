    async imagesUploadHandler(blobInfo, success, failure) {
    const formData = new FormData();
    const file = blobInfo.blob();
    const filename = blobInfo.filename();
    formData.append('contextPath', 'contract_info');
    formData.append('file', file, filename);
    this.uploadingImgCount++;
    try {
        const res = await contractAdd.uploadFile(formData);
        res.data.forEach((item) => {
          success(`${previewUrl}${item.fileId}`);
        });
    } catch (e) {
      failure(e.message, {
        remove: true
      });
    } finally {
      this.uploadingImgCount--;
    }
},

    下面自动处理图片宽高，并且强制修改dom，传给后端的图片宽高就是修改后的，以及点击进行下载

    // 处理图片的宽高
    handlePicture(editor) {
      const observer = new MutationObserver((mutationsList) => {
        for (let mutation of mutationsList) {
          if (mutation.type === 'childList') {
            mutation.addedNodes.forEach(node => {
              if (node.nodeName.toLowerCase() === 'img') {
                this.ensureImageLoadedAndResize(node);
              } else if (node.querySelectorAll) {
                Array.from(node.querySelectorAll('img')).forEach(img => {
                  this.ensureImageLoadedAndResize(img);
                });
              }
            });
          }
        }
      });

      editor.on('init', () => {
        const editorBody = editor.getBody();
        observer.observe(editorBody, { childList: true, subtree: true });
      });
    },
    ensureImageLoadedAndResize(imgElement) {
      const maxWidth = 100;

      const handleImageLoad = () => {
        let newWidth = imgElement.naturalWidth;
        let newHeight = imgElement.naturalHeight;

        if (imgElement.naturalWidth > maxWidth) {
          newWidth = maxWidth;
          newHeight = (imgElement.naturalHeight / imgElement.naturalWidth) * maxWidth;
        }

        // 更新图像宽高和样式
        imgElement.setAttribute('width', newWidth);
        imgElement.setAttribute('height', newHeight);
      };

      // 确保只监听一次加载完成的事件
      const loadListener = () => {
        imgElement.removeEventListener('load', loadListener);
        handleImageLoad();
      };

      // 如果图像已经加载完成，直接处理
      if (imgElement.complete) {
        handleImageLoad();
      } else {
        imgElement.addEventListener('load', loadListener);
      }
    },
